\DoxyCodeLine{30 \textcolor{comment}{// A "{}smart"{} pointer type with reference tracking.  Every pointer to a}}
\DoxyCodeLine{31 \textcolor{comment}{// particular object is kept on a circular linked list.  When the last pointer}}
\DoxyCodeLine{32 \textcolor{comment}{// to an object is destroyed or reassigned, the object is deleted.}}
\DoxyCodeLine{33 \textcolor{comment}{//}}
\DoxyCodeLine{34 \textcolor{comment}{// Used properly, this deletes the object when the last reference goes away.}}
\DoxyCodeLine{35 \textcolor{comment}{// There are several caveats:}}
\DoxyCodeLine{36 \textcolor{comment}{// -\/ Like all reference counting schemes, cycles lead to leaks.}}
\DoxyCodeLine{37 \textcolor{comment}{// -\/ Each smart pointer is actually two pointers (8 bytes instead of 4).}}
\DoxyCodeLine{38 \textcolor{comment}{// -\/ Every time a pointer is assigned, the entire list of pointers to that}}
\DoxyCodeLine{39 \textcolor{comment}{//   object is traversed.  This class is therefore NOT SUITABLE when there}}
\DoxyCodeLine{40 \textcolor{comment}{//   will often be more than two or three pointers to a particular object.}}
\DoxyCodeLine{41 \textcolor{comment}{// -\/ References are only tracked as long as linked\_ptr<> objects are copied.}}
\DoxyCodeLine{42 \textcolor{comment}{//   If a linked\_ptr<> is converted to a raw pointer and back, BAD THINGS}}
\DoxyCodeLine{43 \textcolor{comment}{//   will happen (double deletion).}}
\DoxyCodeLine{44 \textcolor{comment}{//}}
\DoxyCodeLine{45 \textcolor{comment}{// A good use of this class is storing object references in STL containers.}}
\DoxyCodeLine{46 \textcolor{comment}{// You can safely put linked\_ptr<> in a vector<>.}}
\DoxyCodeLine{47 \textcolor{comment}{// Other uses may not be as good.}}
\DoxyCodeLine{48 \textcolor{comment}{//}}
\DoxyCodeLine{49 \textcolor{comment}{// Note: If you use an incomplete type with linked\_ptr<>, the class}}
\DoxyCodeLine{50 \textcolor{comment}{// *containing* linked\_ptr<> must have a constructor and destructor (even}}
\DoxyCodeLine{51 \textcolor{comment}{// if they do nothing!).}}
\DoxyCodeLine{52 \textcolor{comment}{//}}
\DoxyCodeLine{53 \textcolor{comment}{// Bill Gibbons suggested we use something like this.}}
\DoxyCodeLine{54 \textcolor{comment}{//}}
\DoxyCodeLine{55 \textcolor{comment}{// Thread Safety:}}
\DoxyCodeLine{56 \textcolor{comment}{//   Unlike other linked\_ptr implementations, in this implementation}}
\DoxyCodeLine{57 \textcolor{comment}{//   a linked\_ptr object is thread-\/safe in the sense that:}}
\DoxyCodeLine{58 \textcolor{comment}{//     -\/ it's safe to copy linked\_ptr objects concurrently,}}
\DoxyCodeLine{59 \textcolor{comment}{//     -\/ it's safe to copy *from* a linked\_ptr and read its underlying}}
\DoxyCodeLine{60 \textcolor{comment}{//       raw pointer (e.g. via get()) concurrently, and}}
\DoxyCodeLine{61 \textcolor{comment}{//     -\/ it's safe to write to two linked\_ptrs that point to the same}}
\DoxyCodeLine{62 \textcolor{comment}{//       shared object concurrently.}}
\DoxyCodeLine{63 \textcolor{comment}{// FIXME: rename this to safe\_linked\_ptr to avoid}}
\DoxyCodeLine{64 \textcolor{comment}{// confusion with normal linked\_ptr.}}
